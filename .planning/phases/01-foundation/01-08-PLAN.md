    private nonisolated(unsafe) var v2Monitor: Any? = nil

    weak var characterViewModel: CharacterViewModel?
    weak var chatViewModel: ChatViewModel?

    func prepare(character: CharacterViewModel, chat: ChatViewModel) {
        self.characterViewModel = character
        self.chatViewModel = chat
        registerShortcuts()
    }

    func start(_ variant: DemoVariant = .v2) {
        guard !isRunning else { return }
        isRunning = true
        sideLabel = nil

            sideLabel = nil
