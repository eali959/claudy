**CRITICAL RULE — Notification Wiring Stays at Root:**
Every single `.onReceive(NotificationCenter.default.publisher(for: .claudy*))` and every `.onChange(of:)` modifier that affects ViewModel state MUST remain attached to a view in `CharacterRootView.body`. Do NOT move them into sub-views. This is the single most important constraint in this extraction.

```swift
    // etc.

        ZStack {

    }
}
```

**CharacterRootView.swift after extraction:**
```swift
import SwiftUI

struct CharacterRootView: View {
    @State private var characterViewModel = CharacterViewModel()
    @State private var chatViewModel = ChatViewModel()
    @State private var showChat = false
    // ... all @State properties

    var body: some View {
        ZStack {
            CharacterSceneView(characterViewModel: characterViewModel, ...)
            if showChat {
                ChatOverlayView(chatViewModel: chatViewModel, ...)
            }
            PomodoroTimerBadgeContainer(...)
            DemoAnnotationView(...)
        }
        // ALL .onReceive modifiers stay here
        .onReceive(NotificationCenter.default.publisher(for: .claudyToggleChat)) { _ in
            showChat.toggle()
        }
        .onReceive(NotificationCenter.default.publisher(for: .claudyQuickActionFired)) { note in
            // ...
        }
        // ALL .onChange modifiers stay here
        .onChange(of: ...) { ... }
    }
}
```
